        // Function to get history from cookies
        function getHistoryFromCookies() {
            var cookies = document.cookie.split(";");
            for (var i = 0; i < cookies.length; i++) {
                var cookie = cookies[i].trim();
                if (cookie.startsWith("history=")) {
                    return decodeURIComponent(cookie.substring("history=".length));
                }
            }
            return "";
        }

        // Function to add item to cookies
        function addToHistoryCookies(itemText) {
            var history = getHistoryFromCookies();
            var items = history.split(",");
            if (items.length >= 5) {
                // Remove the oldest item if the history is already full
                items.shift();
            }
            items.push(itemText);
            if (items.length > 1) {
                history = items.join(",");
            } else {
                history = itemText;
            }
            setCookie("history", history, 7);
        }

        // Function to remove item from cookies
        function removeFromHistoryCookies(itemText, option_no = "") {
            var history = getHistoryFromCookies();
            var items = history.split(",");
            var index = items.findIndex(function(item) {
                return item.trim() === itemText.trim();
            });

            if (index !== -1) {
                items.splice(index, 1);
                history = items.join(",");
                if (history !== "") {
                    setCookie("history", history, 7);
                } else {
                    deleteCookie("history");
                }

                if (option_no != '') {
                    $('#search_history-' + option_no).remove();
                }
            } else {
                alert("Item not found in history");
            }
        }

        // Function to set a cookie with expiration days
        function setCookie(name, value, days) {
            var expires = "";
            if (days) {
                var date = new Date();
                date.setTime(date.getTime() + days * 24 * 60 * 60 * 1000);
                expires = "; expires=" + date.toUTCString();
            }
            document.cookie = name + "=" + encodeURIComponent(value) + expires + "; path=/";
        }

        // Function to delete a cookie
        function deleteCookie(name) {
            document.cookie = name + "=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;";
        }

        function removeURLParameter(url, parameter) {
            //prefer to use l.search if you have a location/link object
            var urlparts = url.split('?');
            if (urlparts.length >= 2) {

                var prefix = encodeURIComponent(parameter) + '=';
                var pars = urlparts[1].split(/[&;]/g);

                //reverse iteration as may be destructive
                for (var i = pars.length; i-- > 0;) {
                    //idiom for string.startsWith
                    if (pars[i].lastIndexOf(prefix, 0) !== -1) {
                        pars.splice(i, 1);
                    }
                }

                return urlparts[0] + (pars.length > 0 ? '?' + pars.join('&') : '');
            }
            return url;
        }

    let countdownTimer; // Declare countdownTimer in the global scope so it can be accessed to clear the interval

    function startCount() {
        let timeLeft = 120; // Set initial countdown time in seconds (2 minutes = 120 seconds)

        const countdownElement = document.getElementById('countdown');

        // Clear the existing timer if it exists
        if (countdownTimer) {
            clearInterval(countdownTimer);
        }

        countdownTimer = setInterval(function() {
            let minutes = Math.floor(timeLeft / 60); // Calculate minutes
            let seconds = timeLeft % 60; // Calculate remaining seconds

            // Format time to always show 2 digits for seconds
            seconds = seconds < 10 ? '0' + seconds : seconds;

            countdownElement.innerHTML = minutes + ":" + seconds; // Update the countdown display

            if (timeLeft <= 0) {
                clearInterval(countdownTimer); // Stop the timer when it reaches 0
                $('#text-resend').removeClass('d-none')
                $('#text-counter').addClass('d-none')
                $('#btnValidasi').attr('disabled', 'disabled')
            }
            timeLeft -= 1; // Decrease time left by 1 every second
        }, 1000); // Update every second (1000 ms)
    }
